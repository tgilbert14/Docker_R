#### Below is Basic example of what is in Dockerfile ####

# Base R Shiny image
FROM rocker/shiny

# Make a directory in the container
RUN mkdir /home/shiny-app

# Install R dependencies
RUN R -e "install.packages(c('dplyr', 'ggplot2', 'gapminder'))"

# Copy the Shiny app code
COPY app.R /home/shiny-app/app.R

# Expose the application port
EXPOSE 8180

# Run the R Shiny app
CMD Rscript /home/shiny-app/app.R



#### Then commands need to be ran to create ####
#### this one should create the shiny-docker-demo image ####

docker build -t shiny-docker-demo .

#### This one is to run the image inside a container ####

docker run -p 8180:8180 shiny-docker-demo
